大明算学 · 背包动态规划详解

壹 · 01背包

选或不选的经典模型，每个物品仅能选择一次。

状态定义

$$dp[i][j]$$：考虑前 $i$ 个物品，容量为 $j$ 时的最大价值。

转移方程

$$dp[i][j] = \max\left(dp[i-1][j], \; dp[i-1][j-w[i]] + c[i]\right)$$

空间优化后

$$dp[j] = \max\left(dp[j],\;dp[j-w[i]]+c[i]\right)$$（倒序枚举容量）

#include<bits/stdc++.h>
using namespace std;
int dp[202][202],w[10101],c[10101];
int main(){
    int n,m;
    cin>>n>>m;
    swap(n,m);
    for(int i=1;i<=n;i++) cin>>w[i]>>c[i];
    for(int i=1;i<=n;i++){
        for(int j=0;j<=m;j++){
            dp[i][j]=dp[i-1][j];
            if(j>=w[i]) dp[i][j]=max(dp[i][j],dp[i-1][j-w[i]]+c[i]);
        }
    }
    cout<<dp[n][m];
    return 0;
}

贰 · 完全背包

每个物品可选择无限次，核心是正序枚举容量。

转移方程

$$dp[j] = \max(dp[j], dp[j-w[i]] + v[i])$$

#include<iostream>
int d[10121],d2[12912],dp[8322];
using namespace std;
int main(){
    int n,m;
    cin>>m>>n;
    for(int i=1;i<=n;i++) cin>>d[i]>>d2[i];
    for(int i=1;i<=n;i++){
        for(int j=d[i];j<=m;j++){
            dp[j]=max(dp[j],dp[j-d[i]]+d2[i]);
        }
    }
    cout<<"max="<<dp[m];
    return 0;
}

叁 · 多重背包（二进制优化）

物品数量有限，将数量拆分为2的幂次组合，转化为01背包。

for(int i=1;i<=n;i++) cin>>d[i]>>d2[i]>>d3[i];
for(int i=1;i<=n;i++){
    int l=d3[i];
    for(int k=1;k<=l;l-=k,k*=2){
        for(int j=m;j>=k*d[i];j--) 
            dp[j]=max(dp[j],dp[j-k*d[i]]+d2[i]*k);
    }
    for(int j=m;j>=l*d[i];j--) 
        dp[j]=max(dp[j],dp[j-l*d[i]]+d2[i]*l);
}

肆 · 可行性背包

判断是否能恰好装满指定容量的背包，状态为布尔值。

#include<bits/stdc++.h>
using namespace std;
int n,m,a[323232];
bool dp[323232];
int main(){
    cin.tie(0)->sync_with_stdio(0);
    int n,m;
    cin>>n>>m;
    for(int i=1;i<=n;i++) cin>>a[i];
    dp[0]=1;//0容量必然装满
    for(int i=1;i<=n;i++){
        for(int j=m;j>=a[i];j--){
            if(dp[j-a[i]]){
                dp[j]=1;
            }
        }
    }
    if(dp[m]) cout<<"Yes";
    else cout<<"No";
    return 0;
}
